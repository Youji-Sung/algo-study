# 개수 세기
## 성유지

```cpp
#include <iostream>
using namespace std;

int main() {
    int N;
    cin >> N;
    int* arr = new int[N];

    for (int i = 0; i < N; ++i) {
        cin >> arr[i];
    }

    int v;
    cin >> v;

    int res = 0;
    for (int i = 0; i < N; ++i) {
        if (arr[i] == v) {
            res += 1;
        }
    }
    
    cout << res;
}
```
정렬을 해서 풀어보고 싶어요
```cpp
#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int N;
    cin >> N;
    int* arr = new int[N];

    for (int i = 0; i < N; ++i) {
        cin >> arr[i];
    }

    int v;
    cin >> v;

    sort(arr, arr + N);

    int count = upper_bound(arr, arr + N, v) - lower_bound(arr, arr + N, v);

    cout << count << endl;
}
```
algorithm이라는 라이브러리가 있다니!
