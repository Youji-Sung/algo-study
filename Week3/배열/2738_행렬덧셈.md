# 행렬 덧셈

## 전지성

```cpp
#include <iostream>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    int* result = new int[N * M];

    for(int i = 0; i < N * M; i++) {
        cin >> result[i];
    }

    int temp;
    for(int i = 0; i < N * M; i++) {
        cin >> temp;
        result[i] += temp;
    }

    for(int i = 0; i < N; i++) {
        for(int j = 0; j < M; j++) {
            cout << result[i * M + j] << ' ';
        }
        cout << '\n';
    }

    delete[] result;

    return 0;
}
```
