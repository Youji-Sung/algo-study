# 회전하는 큐

## 연예진

```cpp
#include <iostream>
#include <cmath>
#include <vector>

using namespace std;

int main(){
    int N, M, end;
    int start = 1;
    int cnt = 0;
    vector<int> index;

    scanf("%d %d", &N, &M);

    end = N;

    for(int i=0; i<M; i++){
        int num = 0;
        scanf("%d", &num);
        index.push_back(num);
    }

    for(int i=0; i<M; i++){
        if(index.front() != start){
            if(index.front() <= (1 + end) / 2){
                int gap = index.front() - start;

                index.erase(index.begin());
                for(int j=0; j<index.size(); j++){
                    index.at(j) -= gap;
                    if(index.at(j) < start){
                        index.at(j) = end - (start - index.at(j)) + 1;
                    }
                }
                cnt += gap;
            }else{
                int gap = end - index.front() + 1;

                index.erase(index.begin());
                for(int j=0; j<index.size(); j++){
                    index.at(j) += gap;
                    if(index.at(j) > end){
                        index.at(j) = start + index.at(j) - end - 1;
                    }
                }
                cnt += gap;
            }
        }else{
            index.erase(index.begin());
            for(int j=0; j<index.size(); j++){
                index.at(j)--;
            }
        }
        start++;
    }

    printf("%d", cnt);

    return 0;

}
```

## 가승현

```cpp
#include <iostream>
#include <deque>

using namespace std;

int main() {
    int n, m;
    cin >> n >> m;

    deque<int> d;
    for (int i = 1; i <= n; ++i)
        d.push_back(i);

    int rotationCount = 0;

    while (m--) {
        int target;
        cin >> target;
        
        int index = 0;
        while (d[index] != target)
            ++index;
        
        int left = index;
        int right = d.size() - index;
        
        if (left <= right) {
            while (d.front() != target) {
                d.push_back(d.front());
                d.pop_front();
                ++rotationCount;
            }
        } else {
            while (d.front() != target) {
                d.push_front(d.back());
                d.pop_back();
                ++rotationCount;
            }
        }
        d.pop_front();
    }
    
    cout << rotationCount << endl;
    return 0;
}
```
