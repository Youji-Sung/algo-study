# 회전하는 큐

## 연예진

```cpp
#include <iostream>
#include <cmath>
#include <vector>

using namespace std;

int main(){
    int N, M, end;
    int start = 1;
    int cnt = 0;
    vector<int> index;

    scanf("%d %d", &N, &M);

    end = N;

    for(int i=0; i<M; i++){
        int num = 0;
        scanf("%d", &num);
        index.push_back(num);
    }

    for(int i=0; i<M; i++){
        if(index.front() != start){
            if(index.front() <= (1 + end) / 2){
                int gap = index.front() - start;

                index.erase(index.begin());
                for(int j=0; j<index.size(); j++){
                    index.at(j) -= gap;
                    if(index.at(j) < start){
                        index.at(j) = end - (start - index.at(j)) + 1;
                    }
                }
                cnt += gap;
            }else{
                int gap = end - index.front() + 1;

                index.erase(index.begin());
                for(int j=0; j<index.size(); j++){
                    index.at(j) += gap;
                    if(index.at(j) > end){
                        index.at(j) = start + index.at(j) - end - 1;
                    }
                }
                cnt += gap;
            }
        }else{
            index.erase(index.begin());
            for(int j=0; j<index.size(); j++){
                index.at(j)--;
            }
        }
        start++;
    }

    printf("%d", cnt);

    return 0;

}
```
